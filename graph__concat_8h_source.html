66class ConcatStreamGraph : public adf::graph {
67
68 private:
69 static constexpr int L1_LCNT = LCNT / 2;

71 public:
72 adf::vector<adf::kernel> k1;
73 adf::vector<adf::kernel> k2;

75
76 adf::port<input> pin[LCNT];

79 template<int mINP_W1, int mINP_W2, int mOUT_W>
80 adf::kernel create_concat_kernel() {
81 adf::kernel new_k = adf::kernel::create_object<CONCAT_STREAM<TT, H, mINP_W1, mINP_W2, mOUT_W>>();
82 adf::source(new_k) = "concat.cc";
83 adf::headers(new_k) = {"concat.h"};
84 adf::runtime<ratio>(new_k) = 0.1;
85 return new_k;
86 }
87
88 // separate tiles since each tile only 2 input streams
89 ConcatStreamGraph() {
90 static_assert(LCNT <= 8 && LCNT > 1);
91
92 adf::kernel _k;
93 for (int i = 0; i < LCNT-1; i+=2) {
94 _k = create_concat_kernel<INP_W, INP_W, 2*INP_W>();
95 k1.push_back(_k);
96 adf::connect<adf::stream> (pin[i], _k.in[0]);
97 adf::connect<adf::stream> (pin[i+1], _k.in[1]);
98 adf::location<adf::stack> (_k) = adf::location<adf::kernel>(_k);
99 }
100 if (LCNT > 3) { // kernel in loop will be included during compilation otherwise
101 for (int i = 0; i < LCNT-3; i+=4) {
102 _k = create_concat_kernel<2*INP_W, 2*INP_W, 4*INP_W>();
103 k2.push_back(_k);
104 adf::connect<adf::stream> (k1[i/2].out[0], _k.in[0]);
105 adf::connect<adf::stream> (k1[i/2+1].out[0], _k.in[1]);
106 adf::location<adf::stack> (_k) = adf::location<adf::kernel>(_k);
107 }
108 }
109
110
111 if (LCNT == 2) {
112 adf::connect<adf::stream> (k1[0].out[0], pout[0]);
113 adf::samples_per_iteration(k1[0].out[0]) = H*OUT_W;
114 }
115 else if (LCNT == 3) {
116 _k = create_concat_kernel<2*INP_W, INP_W, OUT_W>();
117 k.push_back(_k);
118 adf::connect<adf::stream> (k1[0].out[0], _k.in[0]);
119 adf::connect<adf::stream> (pin[LCNT-1], _k.in[1]);
120
121 adf::connect<adf::stream> (_k.out[0], pout[0]);
122 adf::samples_per_iteration(_k.out[0]) = H*OUT_W;
123 }
124 else if (LCNT == 4) {
125 adf::connect<adf::stream> (k2[0].out[0], pout[0]);
126 adf::samples_per_iteration(k2[0].out[0]) = H*OUT_W;
127 }
128 else if (LCNT == 5) {
129 _k = create_concat_kernel<4*INP_W, INP_W, OUT_W>();
130 k.push_back(_k);
131 adf::connect<adf::stream> (k2[0].out[0], _k.in[0]);
132 adf::connect<adf::stream> (pin[LCNT-1], _k.in[1]);
133
134 adf::connect<adf::stream> (_k.out[0], pout[0]);
135 adf::samples_per_iteration(_k.out[0]) = H*OUT_W;
136 }
137 else if (LCNT == 6) {
138 _k = create_concat_kernel<4*INP_W, 2*INP_W, OUT_W>();
139 k.push_back(_k);
140 adf::connect<adf::stream> (k2[0].out[0], _k.in[0]);
141 adf::connect<adf::stream> (k1[2].out[0], _k.in[1]);
142
143 adf::connect<adf::stream> (_k.out[0], pout[0]);
144 adf::samples_per_iteration(_k.out[0]) = H*OUT_W;
145 }
146 else if (LCNT == 7) {
147 adf::kernel _k1 = create_concat_kernel<2*INP_W, INP_W, 4*INP_W>();
148 k.push_back(_k1);
149 adf::connect<adf::stream> (k1[2].out[0], _k1.in[0]);
150 adf::connect<adf::stream> (pin[LCNT-1], _k1.in[1]);
151
152 adf::kernel _k2 = create_concat_kernel<4*INP_W, 4*INP_W, OUT_W>();
153 k.push_back(_k2);
154 adf::connect<adf::stream> (k2[0].out[0], _k2.in[0]);
155 adf::connect<adf::stream> (_k1.out[0], _k2.in[1]);
156
157 adf::connect<adf::stream> (_k2.out[0], pout[0]);
158 adf::samples_per_iteration(_k2.out[0]) = H*OUT_W;
159 }
160 else if (LCNT == 8) {
161 _k = create_concat_kernel<4*INP_W, 4*INP_W, OUT_W>();
162 k.push_back(_k);
163 adf::connect<adf::stream> (k2[0].out[0], _k.in[0]);
164 adf::connect<adf::stream> (k2[1].out[0], _k.in[1]);

212class ConcatTwiceGraph : public adf::graph {
213
214 public:
215 static constexpr int CONCAT_CNT = (LCNT+7)/8;
216 static constexpr int LCNT_REM = (LCNT % 8 == 0) ? 8 : LCNT % 8;
217 static constexpr int INNER_OUT_W = INP_W * 8;
218 adf::kernel k[CONCAT_CNT];
219 adf::kernel klast;
220 adf::port<input> pin[LCNT];
221 adf::port<output> pout[1];
222
223 ConcatTwiceGraph() {
224 static_assert(LCNT <= 64);
225 klast = adf::kernel::create_object<CONCAT<TT, CONCAT_CNT, H, INNER_OUT_W, OUT_W>>();
226 adf::source(klast) = "concat.cc";
227 adf::headers(klast) = {"concat.h"};
228 adf::runtime<ratio>(klast) = 0.6;
229
230 // intermediate concats
231 if (CONCAT_CNT > 1) { // register kernel error if CONCAT_CNT=1
232 for (int i = 0; i < CONCAT_CNT - 1; i++) {
233 k[i] = adf::kernel::create_object<CONCAT<TT, 8, H, INP_W, INNER_OUT_W>>();
234 adf::source(k[i]) = "concat.cc";
235 adf::headers(k[i]) = {"concat.h"};
236 adf::runtime<ratio>(k[i]) = 0.6;
237
238 for (int j = 0; j < 8; j++)
239 adf::connect<adf::window<H*INP_W*sizeof(TT)>> (pin[i*8+j], k[i].in[j]);
240 adf::connect<adf::stream> (k[i].out[0], klast.in[i]);
241 }
242 }
243
244 int i = CONCAT_CNT - 1;
245 // remainder intermediate concat
246 k[i] = adf::kernel::create_object<CONCAT<TT, LCNT_REM, H, INP_W, INNER_OUT_W>>();
247 adf::source(k[i]) = "concat.cc";
248 adf::headers(k[i]) = {"concat.h"};
249 adf::runtime<ratio>(k[i]) = 0.6;
250
251 for (int j = 0; j < LCNT_REM; j++)
252 adf::connect<adf::window<H*INP_W*sizeof(TT)>> (pin[i*8+j], k[i].in[j]);
253 adf::connect<adf::stream> (k[i].out[0], klast.in[i]);
254
255 adf::connect<adf::stream> (klast.out[0], pout[0]);
256 }
