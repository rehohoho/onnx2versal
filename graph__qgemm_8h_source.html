88class QgemmChunkNStreamGraph : public adf::graph {
89
90 private:
91 adf::relative_coordinate tileOffsets[8] = {
92 {.col_offset = -1, .row_offset = 1}, // top row
93 {.col_offset = 0, .row_offset = 1},
94 {.col_offset = 1, .row_offset = 1},
95 {.col_offset = -1, .row_offset = -1}, // bottom row
96 {.col_offset = 0, .row_offset = -1},
97 {.col_offset = 1, .row_offset = -1},
98 {.col_offset = -1, .row_offset = 0}, // left, right
99 {.col_offset = 1, .row_offset = 0},
100 };
101
102 public:
103 static const int CHUNK_COUNT = (N + NCHUNK - 1) / NCHUNK; // ceiling
104 adf::kernel k[CHUNK_COUNT];
105 ConcatGraph<CONCAT, int8_t, CHUNK_COUNT, M, NCHUNK, N> concat_g;
106
107 adf::port<input> pin[1];
108 adf::port<output> pout[1];
109
110 QgemmChunkNStreamGraph(
111 std::vector<int8_t> weights, // KxN
112 std::vector<int32_t> bias, // N
113 float x_scale,
114 float w_scale,
115 float y_scale,
116 int8_t x_zero,
117 int8_t w_zero,
118 int8_t y_zero,
119 int repeat_cnt = 1
120 ) {
121 static_assert(CHUNK_COUNT <= 8);
122 static_assert(M*K <= TILE_BYTES);
123 static_assert(K*NCHUNK <= MAX_PARAM_BYTES);
124 static_assert(M*NCHUNK <= TILE_BYTES);
125
126 std::vector<int32_t> bChunk;
127
128 for (int i = 0; i < CHUNK_COUNT; i++) {
129
130 // build wchunk
131 std::vector<int8_t> wChunk;
132 wChunk.reserve(NCHUNK*K);
133 for (int j = 0; j < K*N; j+=N) {
134 wChunk.insert(wChunk.end(), weights.begin()+j+i*NCHUNK, weights.begin()+j+i*NCHUNK+NCHUNK);
135 }
136
137 // build bChunk
138 bChunk = std::vector<int32_t>(bias.begin()+i*NCHUNK, bias.begin()+i*NCHUNK+NCHUNK);
139 bChunk.resize(NCHUNK, 0);
140
141 k[i] = adf::kernel::create_object<QGEMM<M, K, NCHUNK>>(
142 wChunk, bChunk, x_scale, w_scale, y_scale, x_zero, w_zero, y_zero);
143 adf::source(k[i]) = "qgemm.cc";
144 adf::headers(k[i]) = {"qgemm.h"};
145 adf::runtime<ratio>(k[i]) = 0.6;
146 adf::repetition_count(k[i]) = repeat_cnt;
147
148 adf::location<adf::kernel>(k[i]) = adf::location<adf::kernel>(concat_g.k[0]) +
149 adf::relative_offset(tileOffsets[i]);
150 adf::location_constraint tilePos = adf::location<adf::kernel>(k[i]);
151 adf::location<adf::parameter>(k[i].param[0]) = tilePos;
152 adf::location<adf::parameter>(k[i].param[0]) = adf::offset(0);
153 adf::location<adf::parameter>(k[i].param[1]) = tilePos;
154 adf::location<adf::parameter>(k[i].param[1]) = adf::offset((K*NCHUNK+31)/32*32);
155 // arbitrary input/output buffer location due to interconnect design
156 }
157
158 for (int i = 0; i < CHUNK_COUNT; i++) {
159 adf::connect<adf::window<M*K>> (pin[0], k[i].in[0]);
160 adf::connect<adf::stream> (k[i].out[0], concat_g.pin[i]);
161 adf::samples_per_iteration(k[i].out[0]) = M*N;
162 }
163 adf::connect<adf::stream> (concat_g.pout[0], pout[0]);
164 }
165
166};
