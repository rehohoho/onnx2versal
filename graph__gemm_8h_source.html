143class GemmReluMknkChunkGraph : public adf::graph {
144
145 private:
146 static const int NCUTCHUNK = N % NCHUNK;
147
148 adf::relative_coordinate tileOffsets[8] = {
149 {.col_offset = -1, .row_offset = 0}, // left, right
150 {.col_offset = 1, .row_offset = 0},
151 {.col_offset = -1, .row_offset = 1}, // bottom row
152 {.col_offset = 0, .row_offset = 1},
153 {.col_offset = 1, .row_offset = 1},
154 {.col_offset = -1, .row_offset = -1}, // top row
155 {.col_offset = 0, .row_offset = -1},
156 {.col_offset = 1, .row_offset = -1},
157 };
158
159 public:
160 static const int CHUNK_COUNT = (N + NCHUNK - 1) / NCHUNK; // ceiling
161 adf::kernel gemms[CHUNK_COUNT];
162 ConcatGraph<CONCAT, float_t, CHUNK_COUNT, 1, NCHUNK, N> concat_g;
163
164 adf::port<input> pin[CHUNK_COUNT];
165 adf::port<output> pout[1];

167 GemmReluMknkChunkGraph(
168 std::vector<float> weights,
169 std::vector<float> bias,
170 int repeat_cnt = 1
171 ) {
172 static_assert(CHUNK_COUNT <= 8);
173 static_assert(M*K*4 <= TILE_BYTES);
174 static_assert(K*N*4 <= MAX_PARAM_BYTES);
175 static_assert(M*N*4 <= TILE_BYTES);
176
177 std::vector<float> wChunk;
178 std::vector<float> bChunk;
179
180 for (int i = 0; i < CHUNK_COUNT; i++) {
181 int chunkSize = (i*NCHUNK + NCHUNK > N) ? NCUTCHUNK : NCHUNK;
182 wChunk = std::vector<float>(weights.begin()+i*NCHUNK*K, weights.begin()+(i*NCHUNK+chunkSize)*K);
183 wChunk.resize(NCHUNK*K, 0);
184 bChunk = std::vector<float>(bias.begin()+i*NCHUNK, bias.begin()+i*NCHUNK+chunkSize);
185 bChunk.resize(NCHUNK, 0);
186
187 gemms[i] = adf::kernel::create_object<GEMM<M, K, NCHUNK, IS_RELU>>(wChunk, bChunk);
188 adf::source(gemms[i]) = "gemm.cc";
189 adf::headers(gemms[i]) = {"gemm.h"};
190 adf::runtime<ratio>(gemms[i]) = 0.6;
191 adf::repetition_count(gemms[i]) = repeat_cnt;
192
193 adf::location<adf::kernel>(gemms[i]) = adf::location<adf::kernel>(concat_g.k[0]) +
194 adf::relative_offset(tileOffsets[i]);
195
196 int offset = 0;
197 adf::location_constraint tilePos = adf::location<adf::kernel>(gemms[i]);
198 adf::location<adf::parameter>(gemms[i].param[0]) = tilePos; // weight (<= 16384B)
199 adf::location<adf::parameter>(gemms[i].param[0]) = adf::offset(offset);
200 adf::location<adf::parameter>(gemms[i].param[1]) = tilePos; // bias (<= 4096B)
201 adf::location<adf::parameter>(gemms[i].param[1]) = adf::offset(offset + NCHUNK*K*4);
202 // arbitrary input/output buffer location due to interconnect design
203 }
204
205 for (int i = 0; i < CHUNK_COUNT; i++) {
206 adf::connect<adf::window<M*K*4>> (pin[i], gemms[i].in[0]);
207 adf::connect<adf::window<M*NCHUNK*4>> (gemms[i].out[0], concat_g.pin[i]);
208 }
209
210 adf::connect<adf::window<M*N*4>> (concat_g.pout[0], pout[0]);
211 }

213};
