115template <
116 template<int, int, int, int> class GEMM,
117 template<int, int, int, int> class CONCAT,
118 int NCHUNK, int M, int K, int N>
119class GemmReluMknkChunkGraph : public adf::graph {
120
121 private:
122 static const int NCUTCHUNK = N % NCHUNK;
123
124 adf::relative_coordinate tileOffsets[8] = {
125 {.col_offset = -1, .row_offset = 0}, // left, right
126 {.col_offset = 1, .row_offset = 0},
127 {.col_offset = -1, .row_offset = 1}, // bottom row
128 {.col_offset = 0, .row_offset = 1},
129 {.col_offset = 1, .row_offset = 1},
130 {.col_offset = -1, .row_offset = -1}, // top row
131 {.col_offset = 0, .row_offset = -1},
132 {.col_offset = 1, .row_offset = -1},
133 };
134
135 public:
136 static const int CHUNK_COUNT = (N + NCHUNK - 1) / NCHUNK; // ceiling
137 adf::kernel gemms[CHUNK_COUNT];
138 ConcatGraph<CONCAT, CHUNK_COUNT, 1, NCHUNK, N> concat_g;
139
140 adf::port<input> pin[CHUNK_COUNT];
141 adf::port<output> pout[1];
142
143 GemmReluMknkChunkGraph(
144 std::vector<float> weights,
145 std::vector<float> bias
146 ) {
147 std::vector<float> wChunk;
148 std::vector<float> bChunk;
149
150 for (int i = 0; i < CHUNK_COUNT; i++) {
151 int chunkSize = (i*NCHUNK + NCHUNK > N) ? NCUTCHUNK : NCHUNK;
152 wChunk = std::vector<float>(weights.begin()+i*NCHUNK*K, weights.begin()+(i*NCHUNK+chunkSize)*K);
153 wChunk.resize(NCHUNK*K, 0);
154 bChunk = std::vector<float>(bias.begin()+i*NCHUNK, bias.begin()+i*NCHUNK+chunkSize);
155 bChunk.resize(NCHUNK, 0);
156
157 gemms[i] = adf::kernel::create_object<GEMM<M, K, NCHUNK, NCHUNK>>(wChunk, bChunk);
158 adf::source(gemms[i]) = "gemm.cc";
159 adf::headers(gemms[i]) = {"gemm.h"};
160 adf::runtime<ratio>(gemms[i]) = 0.6;
161
162 adf::location<adf::kernel>(gemms[i]) = adf::location<adf::kernel>(concat_g.k[0]) +
163 adf::relative_offset(tileOffsets[i]);
164 adf::location_constraint tilePos = adf::location<adf::kernel>(gemms[i]);
165 adf::location<adf::parameter>(gemms[i].param[0]) = tilePos; // weight (<= 16384B)
166 adf::location<adf::parameter>(gemms[i].param[0]) = adf::offset(0x0000);
167 adf::location<adf::parameter>(gemms[i].param[1]) = tilePos; // bias (<= 4096B)
168 adf::location<adf::parameter>(gemms[i].param[1]) = adf::offset(0x4000);
169 adf::location<adf::buffer>(gemms[i].in[0]) = tilePos; // input window (<= 4096B)
170 adf::location<adf::buffer>(gemms[i].in[0]) = {adf::offset(0x5000), adf::offset(0x6000)};
171 }
172
173 for (int i = 0; i < CHUNK_COUNT; i++) {
174 adf::connect<adf::window<M*K*4>> (pin[i], gemms[i].in[0]);
175 adf::connect<adf::window<M*NCHUNK*4>> (gemms[i].out[0], concat_g.pin[i]);
176 }
177
178 adf::connect<adf::window<M*N*4>> (concat_g.pout[0], pout[0]);
179 }
180
181};
