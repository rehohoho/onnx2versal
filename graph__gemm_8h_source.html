110template <
111 template<int, int, int> class GEMM,
112 template<int, int, int, int> class CONCAT,
113 int NCHUNK, int M, int K, int N>
114class GemmReluMknkChunkGraph : public adf::graph {
115
116 private:
117 static const int NCUTCHUNK = N % NCHUNK;
118
119 adf::relative_coordinate tileOffsets[8] = {
120 {.col_offset = -1, .row_offset = 0}, // left, right
121 {.col_offset = 1, .row_offset = 0},
122 {.col_offset = -1, .row_offset = 1}, // bottom row
123 {.col_offset = 0, .row_offset = 1},
124 {.col_offset = 1, .row_offset = 1},
125 {.col_offset = -1, .row_offset = -1}, // top row
126 {.col_offset = 0, .row_offset = -1},
127 {.col_offset = 1, .row_offset = -1},
128 };
129
130 public:
131 static const int CHUNK_COUNT = (N + NCHUNK - 1) / NCHUNK; // ceiling
132 adf::kernel gemms[CHUNK_COUNT];
133 ConcatGraph<CONCAT, CHUNK_COUNT, NCHUNK, NCHUNK, N> concat_g;
134
135 adf::port<input> pin[CHUNK_COUNT];
136 adf::port<output> pout[1];
137
138 GemmReluMknkChunkGraph(
139 std::vector<float> weights,
140 std::vector<float> bias
141 ) {
142 std::vector<float> wChunk;
143 std::vector<float> bChunk;
144
145 for (int i = 0; i < CHUNK_COUNT; i++) {
146 int chunkSize = (i*NCHUNK + NCHUNK > N) ? NCUTCHUNK : NCHUNK;
147 wChunk = std::vector<float>(weights.begin()+i*NCHUNK*K, weights.begin()+(i*NCHUNK+chunkSize)*K);
148 wChunk.resize(NCHUNK*K, 0);
149 bChunk = std::vector<float>(bias.begin()+i*NCHUNK, bias.begin()+i*NCHUNK+chunkSize);
150 bChunk.resize(NCHUNK, 0);
151 gemms[i] = adf::kernel::create_object<GEMM<M, K, NCHUNK>>(wChunk, bChunk);
152 adf::source(gemms[i]) = "gemm.cc";
153 adf::runtime<ratio>(gemms[i]) = 0.6;
154
155 adf::location<adf::kernel>(gemms[i]) = adf::location<adf::kernel>(concat_g.k[0]) +
156 adf::relative_offset(tileOffsets[i]);
157 adf::location_constraint tilePos = adf::location<adf::kernel>(gemms[i]);
158 adf::location<adf::parameter>(gemms[i].param[0]) = tilePos; // weight (<= 16384B)
159 adf::location<adf::parameter>(gemms[i].param[0]) = adf::offset(0x0000);
160 adf::location<adf::parameter>(gemms[i].param[1]) = tilePos; // bias (<= 4096B)
161 adf::location<adf::parameter>(gemms[i].param[1]) = adf::offset(0x4000);
162 adf::location<adf::buffer>(gemms[i].in[0]) = tilePos; // input window (<= 4096B)
163 adf::location<adf::buffer>(gemms[i].in[0]) = {adf::offset(0x5000), adf::offset(0x6000)};
164 }
165
166 for (int i = 0; i < CHUNK_COUNT; i++) {
167 adf::connect<adf::window<M*K*4>> (pin[i], gemms[i].in[0]);
168 adf::connect<adf::window<M*NCHUNK*4>> (gemms[i].out[0], concat_g.pin[i]);
169 }
170
171 adf::connect<adf::window<M*N*4>> (concat_g.pout[0], pout[0]);
172 }
173
174};
