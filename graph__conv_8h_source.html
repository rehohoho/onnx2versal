125template <
126 template<int, int, int, int, int, int> class CONV,
127 template<int, int, int, int> class CONCAT,
128 int IS_BCHW, int IS_KPAD,
129 int MCHUNK, int INP_W, int OUT_W, int B, int C, int M, int K>
130class ConvReluChunkGraph : public adf::graph {
131
132 private:
133 static const int K2 = (IS_KPAD) ? (K+7)/8*8 : K;
134 static const int MCUTCHUNK = M % MCHUNK;
135 static const int CONCAT_CHUNK = (IS_BCHW) ? MCHUNK*OUT_W*OUT_W : MCHUNK;
136 static const int CONCAT_BLOCK = (IS_BCHW) ? M*OUT_W*OUT_W : M;
137 static const int CONCAT_CHUNK_COUNT = B*MCHUNK*OUT_W*OUT_W / CONCAT_CHUNK;
138
139 adf::relative_coordinate tileOffsets[8] = {
140 {.col_offset = -1, .row_offset = 0}, // left, right
141 {.col_offset = 1, .row_offset = 0},
142 {.col_offset = -1, .row_offset = 1}, // bottom row
143 {.col_offset = 0, .row_offset = 1},
144 {.col_offset = 1, .row_offset = 1},
145 {.col_offset = -1, .row_offset = -1}, // top row
146 {.col_offset = 0, .row_offset = -1},
147 {.col_offset = 1, .row_offset = -1},
148 };
149
150 public:
151 static const int CHUNK_COUNT = (M + MCHUNK - 1) / MCHUNK; // ceiling
152 adf::kernel convs[CHUNK_COUNT];
153 ConcatGraph<CONCAT, CHUNK_COUNT, CONCAT_CHUNK_COUNT, CONCAT_CHUNK, CONCAT_BLOCK> concat_g;
154
155 adf::port<input> pin[CHUNK_COUNT];
156 adf::port<output> pout[1];
157
158 ConvReluChunkGraph(
159 std::vector<float> weights,
160 std::vector<float> bias
161 ) {
162 std::vector<float> wChunk;
163 std::vector<float> bChunk;
164
165 for (int i = 0; i < CHUNK_COUNT; i++) {
166 int chunkSize = (i*MCHUNK + MCHUNK > M) ? MCUTCHUNK : MCHUNK;
167 wChunk = std::vector<float>(weights.begin()+i*MCHUNK*K*K2*C,
168 weights.begin()+(i*MCHUNK+chunkSize)*K*K2*C);
169 wChunk.resize(MCHUNK*K*K2*C, 0);
170 bChunk = std::vector<float>(bias.begin()+i*MCHUNK, bias.begin()+i*MCHUNK+chunkSize);
171 bChunk.resize(MCHUNK, 0);
172
173 convs[i] = adf::kernel::create_object<CONV<INP_W, OUT_W, B, C, MCHUNK, K>>(wChunk, bChunk);
174 adf::source(convs[i]) = "conv.cc";
175 adf::headers(convs[i]) = {"conv.h"};
176 adf::runtime<ratio>(convs[i]) = 0.6;
177
178 adf::location<adf::kernel>(convs[i]) = adf::location<adf::kernel>(concat_g.k[0]) +
179 adf::relative_offset(tileOffsets[i]);
180 adf::location_constraint tilePos = adf::location<adf::kernel>(convs[i]);
181 adf::location<adf::parameter>(convs[i].param[0]) = tilePos; // weight (<= 16384B)
182 adf::location<adf::parameter>(convs[i].param[0]) = adf::offset(0x0000);
183 adf::location<adf::parameter>(convs[i].param[1]) = tilePos; // bias (<= 4096B)
184 adf::location<adf::parameter>(convs[i].param[1]) = adf::offset(0x4000);
185 // input window and output window can be much larger
186 }
187
188 for (int i = 0; i < CHUNK_COUNT; i++) {
189 adf::connect<adf::window<B*INP_W*INP_W*C*4>> (pin[i], convs[i].in[0]);
190 adf::connect<adf::window<B*OUT_W*OUT_W*MCHUNK*4>> (convs[i].out[0], concat_g.pin[i]);
191 }
192 adf::connect<adf::window<B*OUT_W*OUT_W*M*4>> (concat_g.pout[0], pout[0]);
193 }
194};
